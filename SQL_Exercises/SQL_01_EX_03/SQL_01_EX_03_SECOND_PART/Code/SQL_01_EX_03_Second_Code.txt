SELECT DISTINCT SUBSTR(PHONE_NUMBER,1,3) FROM HR.EMPLOYEES;
SELECT  COUNT(CASE WHEN SUBSTR(PHONE_NUMBER, 1, 3) = '515' THEN 1 END) AS GSM_515,
        COUNT(CASE WHEN SUBSTR(PHONE_NUMBER, 1, 3) = '603' THEN 1 END) AS GSM_603,
        COUNT(CASE WHEN SUBSTR(PHONE_NUMBER, 1, 3) = '650' THEN 1 END) AS GSM_650,
        COUNT(CASE WHEN SUBSTR(PHONE_NUMBER, 1, 3) = '011' THEN 1 END) AS GSM_011,
        COUNT(CASE WHEN SUBSTR(PHONE_NUMBER, 1, 3) = '590' THEN 1 END) AS GSM_590
            FROM HR.EMPLOYEES;
